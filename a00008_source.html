104 CameraState ClampBounds(CameraState state, float cameraAspect)
105 {
106 Bounds bounds = state.GetBounds(cameraAspect);
107 Vector3 correction = Vector3.zero;
108 if (clampXAxis){
109 float xCorrection = GetCorrection(bounds.min.x, bounds.max.x, xClamp.x, xClamp.y);
110 correction += new Vector3(xCorrection, 0, 0);
111 }
112
113 if (clampYAxis){
114 float yCorrection = GetCorrection(bounds.min.y, bounds.max.y, yClamp.x, yClamp.y);
115 correction += new Vector3(0, yCorrection, 0);
116 }
117
118 bounds.center += correction;
119 bounds.max = Vector3.Min(bounds.max, new Vector3(xClamp.y, yClamp.y, 0));
120 bounds.min = Vector3.Max(bounds.min, new Vector3(xClamp.x, yClamp.x, 0));
121 Vector2 correctionSign = new Vector2(correction.x == 0 ? 0 : Mathf.Sign(correction.x),
122 correction.y == 0 ? 0 : Mathf.Sign(correction.y));
123 Vector2 pivot = new Vector2(0.5f, 0.5f) - correctionSign * 0.5f;
124
125 return CameraState.ContainingBounds(bounds, cameraAspect, pivot);
126 }
135 float GetCorrection(float min, float max, float clampMin, float clampMax)
136 {
137 //positive when correction is needed
138 float minCorrection = Mathf.Max(clampMin - min, 0);
139
140 //negative when correction is needed
141 float maxCorrection = Mathf.Min(clampMax - max, 0);
142
143 //Selecting the biggest correction
144 return (minCorrection > -maxCorrection) ? minCorrection : maxCorrection;
145 }
