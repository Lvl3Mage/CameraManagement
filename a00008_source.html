130 public static CameraState CoveringBounds(Bounds bounds, float aspectRatio, Vector2 pivot)
131 {
132 Vector2 pos = bounds.center;
133 Vector2 offset = Vector2.zero;
134 float sizeFromX = bounds.size.x / aspectRatio;
135 float sizeFromY = bounds.size.y;
136 float size = Mathf.Max(sizeFromX, sizeFromY);
137 if(sizeFromX < sizeFromY){
138 offset.x = (bounds.size.x - size*aspectRatio) * ( pivot.x - 0.5f);
139 }
140 else{
141 offset.y = (bounds.size.y - size) * (pivot.y - 0.5f);
142 }
143 pos += offset;
144 return Empty().WithPosition(pos).WithZoom(size*0.5f);
145 }

473 public CameraState ZoomTowards(float targetZoom, Vector2 target)
474 {
475 if (zoom == null){
476 return Clone();
477 }
478 CameraState newState = Clone();
479
480 if(position != null){
481 Vector2 offset = target - position.Value;
482 Vector2 newOffset = offset * (targetZoom / newState.zoom.Value);
483
484 newState.position += offset - newOffset;
485 }
486 newState.zoom = targetZoom;
487 return newState;
488 }
