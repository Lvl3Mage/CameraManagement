134 public static CameraState CoveringBounds(Bounds bounds, float aspectRatio, Vector2 pivot)
135 {
136 Vector2 pos = bounds.center;
137 Vector2 offset = Vector2.zero;
138 float sizeFromX = bounds.size.x / aspectRatio;
139 float sizeFromY = bounds.size.y;
140 float size = Mathf.Max(sizeFromX, sizeFromY);
141 if(sizeFromX < sizeFromY){
142 offset.x = (bounds.size.x - size*aspectRatio) * ( pivot.x - 0.5f);
143 }
144 else{
145 offset.y = (bounds.size.y - size) * (pivot.y - 0.5f);
146 }
147 pos += offset;
148 return Empty().WithPosition(pos).WithZoom(size*0.5f);
149 }

484 public CameraState ZoomTowards(float targetZoom, Vector2 target)
485 {
486 if (zoom == null){
487 return Clone();
488 }
489 CameraState newState = Clone();
490
491 if(position != null){
492 Vector2 offset = target - position.Value;
493 Vector2 newOffset = offset * (targetZoom / newState.zoom.Value);
494
495 newState.position += offset - newOffset;
496 }
